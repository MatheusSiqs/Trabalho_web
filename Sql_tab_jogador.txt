CREATE TABLE `jogo_memoria`.`jogador` (`idJogador` SMALLINT NOT NULL AUTO_INCREMENT COMMENT 'Identificação do Jogador' , `Nome` VARCHAR(30) NOT NULL COMMENT 'Nome Completo' , `DtNasc` DATE NOT NULL COMMENT 'Data de Nascimento' , `CPF` TEXT NOT NULL COMMENT 'CPF' , `Telefone` TEXT NOT NULL COMMENT 'Telefone' , `email` VARCHAR(30) NOT NULL COMMENT 'Email' , `User` VARCHAR(20) NOT NULL COMMENT 'Nome de User' , `Senha` VARCHAR(20) NOT NULL COMMENT 'Senha' , PRIMARY KEY (`idJogador`), UNIQUE `CPF` (`CPF`)) ENGINE = InnoDB COMMENT = 'Tabela de Jogador';

CREATE TABLE `jogo_memoria`.`partidas` (`idPartida` SMALLINT NOT NULL AUTO_INCREMENT COMMENT 'Identificação da partida' , `idJogador` SMALLINT NOT NULL COMMENT 'Identificação do Jogador' , `Tempo` TIME NULL COMMENT 'Duração da partida' , `Pontuação` INT NOT NULL COMMENT 'Pontuação' , `Modo` CHAR(1) NOT NULL COMMENT 'Modo de jogo' , `Tamanho` INT NOT NULL COMMENT 'Tamanho do tabuleiro' , `Status` CHAR(1) NOT NULL COMMENT 'Resultado da partida' , `nJogadas` INT NOT NULL COMMENT 'Numero de jogadas' , `Data` DATETIME NOT NULL COMMENT 'Data' , PRIMARY KEY (`idPartida`)) ENGINE = InnoDB COMMENT = 'Tabela de partidas';
